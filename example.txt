// structure Abstruction

Profile o :
  first // declare public

nameGettable Profile o :
  name : (first ' ' last)

luke : Profile
  first  : 'Luke'
  last   : 'Skywalker'
  age    : 18
  father : Profile
    first  : 'Anakin'
    last


say luke name 

say ^x -> luke[x] 'first'

say luke age //none

say luke father name //none


//Enumerate

  'Luke' : 'first'
  'Skywalker' : 'last'
  18 : 'age'
  'father' : Profile
    first : 'Anakin'
